  <script>
    // --- Dados de exemplo (substitua por dados reais vindos da sua API) ---
    const produtos = Array.from({length:36}).map((_,i)=>({
      id:i+1,
      titulo:`Produto Premium ${i+1}`,
      descricao:"Descrição curta e objetiva do produto, com destaque para o benefício principal.",
      preco:(19 + (i%9)*7) + .90,
      precoOld: (i%3===0) ? (29 + (i%9)*7) + .90 : null,
      rating:(4 + (i%10)/10).toFixed(1),
      categoria: (i%4===0? 'promocoes' : i%5===0? 'maisvendidos' : i%6===0 ? 'novidades' : 'all'),
      img:`https://picsum.photos/seed/p${i+10}/600/450`
    }));

    const state = { page:1, perPage:9, query:"", chip:"all" };

    function setChip(el, chip){
      state.chip = chip; state.page = 1; document.querySelectorAll('.chip').forEach(c=>c.classList.remove('active')); el.classList.add('active');
      render();
    }
    function applyFilters(){ state.query = document.getElementById('q').value.trim().toLowerCase(); state.page=1; render(); }

    function filtered(){
      return produtos.filter(p=>{
        const byChip = state.chip==='all' ? true : p.categoria===state.chip;
        const byQ = state.query? (p.titulo.toLowerCase().includes(state.query) || p.descricao.toLowerCase().includes(state.query)) : true;
        return byChip && byQ;
      })
    }

    function render(){
      const grid = document.getElementById('grid');
      const items = filtered();
      const start = (state.page-1)*state.perPage;
      const pageItems = items.slice(start, start + state.perPage);

      grid.innerHTML = pageItems.map(p=>`
        <article class="card">
          <div class="thumb">
            <img src="${p.img}" alt="${p.titulo}" />
            ${p.categoria==='promocoes' ? '<span class="tag">-20%</span>' : ''}
          </div>
          <div class="content">
            <div class="title">${p.titulo}</div>
            <div class="desc">${p.descricao}</div>
            <div class="price-row">
              <div>
                <span class="price">US$ ${p.preco.toFixed(2)}</span>
                ${p.precoOld? `<span class="old">US$ ${p.precoOld.toFixed(2)}</span>`: ''}
              </div>
              <div class="rating-row">★ ${p.rating}</div>
            </div>
          </div>
          <div class="actions">
            <button class="btn-sm">Adicionar ao carrinho</button>
            <button class="btn-sm primary">Comprar</button>
          </div>
        </article>
      `).join('');

      // Paginação
      const pages = Math.max(1, Math.ceil(items.length / state.perPage));
      const pag = document.getElementById('pagination');
      pag.innerHTML = '';
      for(let i=1;i<=pages;i++){
        const b = document.createElement('button');
        b.textContent = i; b.className = (i===state.page? 'active':'');
        b.addEventListener('click',()=>{ state.page = i; render(); });
        pag.appendChild(b);
      }
    }

    render();
  </script>
