npm i react-multi-carousel

// BannerCarousel.tsx
import React from 'react';
import Carousel from 'react-multi-carousel';
import 'react-multi-carousel/lib/styles.css';

type Banner = { id: string | number; imageUrl: string; title?: string; linkUrl?: string };

const responsive = {
  desktop: { breakpoint: { max: 3000, min: 1280 }, items: 1, partialVisibilityGutter: 48 },
  tablet:  { breakpoint: { max: 1280, min: 768 },  items: 1, partialVisibilityGutter: 36 },
  mobile:  { breakpoint: { max: 768,  min: 0 },    items: 1, partialVisibilityGutter: 28 },
} as const;

export default function BannerCarousel({
  banners,
  autoPlayInterval = 4000,
  showDots = true,
  peekPx = 48,
}: {
  banners: Banner[];
  autoPlayInterval?: number;
  showDots?: boolean;
  peekPx?: number;
}) {
  if (!banners?.length) return null;

  // opcional: sobrescrever gutter via prop
  const resp = {
    desktop: { ...responsive.desktop, partialVisibilityGutter: peekPx },
    tablet:  { ...responsive.tablet,  partialVisibilityGutter: Math.max(32, peekPx - 12) },
    mobile:  { ...responsive.mobile,  partialVisibilityGutter: Math.max(24, peekPx - 20) },
  };

  return (
    <div style={{ position: 'relative', width: '100%' }}>
      <Carousel
        responsive={resp}
        infinite
        autoPlay
        autoPlaySpeed={autoPlayInterval}
        partialVisible
        swipeable
        draggable
        keyBoardControl
        pauseOnHover
        arrows
        showDots={showDots}
        containerClass="banner-carousel-container"
        itemClass="banner-carousel-item"
        sliderClass="banner-carousel-slider"
      >
        {banners.map((b) => (
          <div key={b.id}>
            <div
              onClick={() => b.linkUrl && window.open(b.linkUrl, '_blank')}
              style={{
                width: '100%',
                aspectRatio: '16 / 9', // ou troque por height: 320
                position: 'relative',
                borderRadius: 12,
                overflow: 'hidden',
                boxShadow: '0 8px 24px rgba(0,0,0,0.15)',
                cursor: b.linkUrl ? 'pointer' : 'default',
                background: '#eee',
              }}
            >
              <img
                src={b.imageUrl}
                alt={b.title ?? 'banner'}
                style={{ width: '100%', height: '100%', objectFit: 'cover', display: 'block' }}
                loading="lazy"
                decoding="async"
                draggable={false}
              />
            </div>
          </div>
        ))}
      </Carousel>

      <style>{`
        .banner-carousel-container { overflow: visible; }
        .banner-carousel-slider { overflow: visible; }
        .banner-carousel-item   { padding-right: 0; }
      `}</style>
    </div>
  );
}
