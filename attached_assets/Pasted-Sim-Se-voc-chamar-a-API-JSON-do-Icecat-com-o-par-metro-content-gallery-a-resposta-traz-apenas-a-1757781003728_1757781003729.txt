import fs from "node:fs/promises";
import fetch from "node-fetch";

const API = "https://live.icecat.biz/api";
const HEADERS = {
  "api_token": process.env.ICECAT_API_TOKEN,
  "content_token": process.env.ICECAT_CONTENT_TOKEN,
};

async function getImageUrlsByGTIN(gtin) {
  const url = `${API}?lang=PT&shopname=${process.env.ICECAT_USER}&GTIN=${gtin}&content=gallery`;
  const r = await fetch(url, { headers: HEADERS });
  if (!r.ok) throw new Error(`${r.status} ${await r.text()}`);
  const json = await r.json();

  // O array pode vir em json.Gallery ou json.data.Gallery (depende do plano/endpoint)
  const gallery = json.Gallery || json?.data?.Gallery || [];
  // Retorna s√≥ URLs (maior primeiro)
  return gallery.map(g => g.Pic).filter(Boolean);
}

async function downloadMainImage(gtin, outPath) {
  const urls = await getImageUrlsByGTIN(gtin);
  if (!urls.length) throw new Error("Sem imagens para este GTIN");
  const img = await fetch(urls[0]);
  const buf = Buffer.from(await img.arrayBuffer());
  await fs.writeFile(outPath, buf);
}
